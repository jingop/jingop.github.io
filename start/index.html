# 开始

# 全局安装 vuepress

npm install -g vuepress

# 创建一个文件夹

mkdir 文件名 如 : mkdir personalblog

# 初始化

npm init -y

# 创建文件夹和文件 参考下面的目录

 .
   ├─ docs 文件夹
   │ ├─ README.md
   │ └─ .vuepress 文件夹
   │ └─ config.js
   └─ package.json

# 配置 package.json 的命令

"scripts": {
	"dev": "vuepress dev docs",
	"build": "vuepress build docs"
}

# 配置默认主题 docs/README.md

---
home: true
heroImage: /hero.png
actionText: 快速上手 →
actionLink: /zh/guide/
features:
  - title: 简洁至上
    details: 以 Markdown 为中心的项目结构，以最少的配置帮助你专注于写作。
  - title: Vue驱动
    details: 享受 Vue + webpack 的开发体验，在 Markdown 中使用 Vue 组件，同时可以使用 Vue 来开发自定义主题。
  - title: 高性能
    details: VuePress 为每个页面预渲染生成静态的 HTML，同时在页面被加载的时候，将作为 SPA 运行。
footer: MIT Licensed | Copyright © 2018-present Evan You
---

# 建立静态资源目录

.vuepress/public

（注意vuepress前面的小点，必须加上）

# 在 docs/.vuepress/config.js

module.exports = {
  title: 'Hello VuePress',
  description: 'Just playing around'
  <!-- 此处添加导航 themeConfig-->
};

# 让导航可以使用

在docs下创建一个web文件夹     

docs/web/README.md

# 配置导航 .vuepress/config.js,

 themeConfig: {
  nav: [
    { text: '首页', link: '/' },
    { text: 'web', link: '/web/' },
	 <!-- 此处添加下拉导航-->
  ],
}

# 配置下拉导航 （items）

  • docs/jscj/README.md
  • docs/jszj/README.md
  • docs/jsgj/README.md
{
  text:'js',
  items:[
	  {text:'js初级',link:'/jscj/'},
	  {text:'js中级',link:'/jszj/'},
	  {text:'js高级',link:'/jsgj/'}
  ]
}

# 一级的侧边栏配置 对应的路径就是 docs 下面的文件夹

sidebar: [
  {
	title: 'vueepress',
	collapsable: true, //是否折叠侧边栏
	 children: ['/introduce/', '/start/'，]
  }
]

# 运行命令

npm run dev
